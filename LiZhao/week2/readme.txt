#200个无重复激活码
import random
s1 = 'abcdefghijklmnopqrstuvwxyz'
s2 = s1.upper()
s3 = '0123456789'*3
l = [s1,s2,s3]
for i in range(200):
        count = random.randint(7,7)
        for j in range(0,count):
            ran1 = random.randint(0,2)
            ran2 = random.randint(0,25)
            print(l[ran1][ran2],end='')
        print()
